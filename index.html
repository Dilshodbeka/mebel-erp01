🎨 Аналитика покраски

Статус заказов в покраске

⚠️ Таблицы в Supabase не найдены
Скопируйте и выполните этот SQL в Supabase → SQL Editor:
create table paint_catalog (
  id bigserial primary key,
  category text, name text, area_m2 numeric default 0
);
create table paint_order_items (
  id bigserial primary key,
  order_id text, category text, item_name text, qty integer
);
create table paint_layer_config (
  order_id text primary key,
  layers integer default 2, coats integer default 3
);
create table paint_records (
  id bigserial primary key,
  order_id text, stage_key text, category text,
  item_name text, qty_done integer, worker text,
  created_at timestamptz default now()
);

-- Разрешаем доступ (отключаем RLS)
alter table paint_catalog disable row level security;
alter table paint_order_items disable row level security;
alter table paint_layer_config disable row level security;
alter table paint_records disable row level security;
